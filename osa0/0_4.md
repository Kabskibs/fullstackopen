    deactivate Browser
    activate Server

    deactivate Server
    activate Browser
    Browser->>Server: GET https://studies.cs.helsinki.fi/exampleapp/main.js
    deactivate Browser
    activate Server
    Server-->>Browser: JS file
    deactivate Server
    activate Browser

    Browser->>Server: GET https://studies.cs.helsinki.fi/exampleapp/data.json
    deactivate Browser
    activate Server
